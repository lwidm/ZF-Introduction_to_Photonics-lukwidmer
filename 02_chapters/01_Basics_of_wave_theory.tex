\hrule

		magnetic permeability: & $\mu_0=4\pi\textsc{e}-7 \unit[2]{\frac{H}{m}}$                                    \\
		wavelength:            & $\lambda_0 \unit[2]{m}$                                                           \\
		wave number:           & $k=\frac{2\pi}{\lambda_0}$                                                        \\

		Temporal period:       & $T = \frac{\lambda_0}{v}$                                                         \\
		frequency:             & $f= \frac{1}{f}$                                                                  \\
		Angular frequency:     & $\omega = 2\pi f$                                                                 \\
		plane wave:            & $\Psi(\vv{r},t) = Ae^{i(\vv{k}\cdot \vv{r} - \omega t + \varphi)}$                \\
		uncertainty:           & $\Delta x \Delta k_x \geq \frac{1}{2}$                                            \\
		speed of light (vac):  & $c = \frac{1}{\sqrt{\mu_0 \cdot \varepsilon_0}} = 299.792\textsc{e}6 \unit[2]{m}$ \\
		speed of light:        & $c = \frac{1}{\sqrt{\mu \cdot \varepsilon}} \unit[2]{m}$
	\end{tabular}\newline
}

\vspace{2mm}
\normalsize\textbf{Differential wave equation for 1D system}\small
\hrule
\vspace{2pt}
\myBox{
	$$ \nabla^2 \vv{\Psi} = \frac{1}{v^2} \frac{\partial^2\vv{\Psi}}{\partial t^2} \hspace{2em}\text{or}\hspace{2em} \nabla^2 \vv{\Psi} - \frac{1}{v^2} \frac{\partial^2\vv{\Psi}}{\partial t^2} = 0$$
}
with 1D version:
$$ \frac{\partial^2\Psi}{\partial x^2} = \frac{1}{v^2} \frac{\partial^2\Psi}{\partial t^2} \hspace{2em}\text{or}\hspace{2em} \nabla^2 \Psi - \frac{1}{v^2} \frac{\partial^2\Psi}{\partial t^2} = 0$$

\vspace{2mm}
\normalsize\textbf{Helmholtz equation}\small
\hrule
\vspace{2pt}
\myBox{
	$$ (\nabla^2 + \vv{k}^2)\vv{X}(\vv{x}) = 0 $$
}
with $\vv{\Psi}(\vv{x},t) = \vv{A}e^{i(\vv{k}\vv{x}-\omega t + \varphi)} = \underbrace{\vv{A}e^{i(\vv{k}\vv{x}+\varphi)}}_{\vv{X}(\vv{x})}\underbrace{e^{-i\omega t}}_{T(t)} = \vv{X}(\vv{x})T(t)$ and $k^2 = \frac{\omega^2}{v^2}$

\mySubsection{Fourier Decomposition}{3}{
the 2D Fourier transform $\mathbf{\hat E}$ of the electric field $\mathbf{E}$:
\scriptsize
$$ \mathbf{\hat E}(k_x, k_y, z) = \frac{1}{2\pi} \iint_{-\infty}^\infty \mathbf{E}(x,y,z)e^{-i[k_x x + k_y y]} \mathrm{d}x\mathrm{d}y$$
\small
Similarly, the inverse Fourier transform reads as:
\scriptsize
$$ \mathbf{E}(x, y, z) = \frac{1}{2\pi} \iint_{-\infty}^\infty \mathbf{\hat E}(k_x,k_y,z)e^{i[k_x x + k_y y]} \mathrm{d}k_x\mathrm{d}k_y$$
\small
Assuming $\mathbf{E}(r)$ is time-harmonic\newline
\scriptsize(i.e. $\mathbf{\hat E}(k_x, k_y, z) = \mathbf{\hat E}(k_x,k_y,0) e^{\pm ik_zz}$ )\small
\myBox{

	\scriptsize
	\begin{align*}
		 & \mathbf{E}(x, y, z) =                                                                                                          \\
		 & \frac{1}{2\pi} \iint_{-\infty}^\infty \mathbf{\hat E}(k_x,k_y,0) e^{i[k_x x + k_y y]} e^{\pm ik_zz} \mathrm{d}k_x\mathrm{d}k_y
	\end{align*}
	\small
}
Fourier decomposition of the scattered field at $z = 0$:
$$\mathbf{\hat E}(k_x,k_y,0) e^{i[k_x x + k_y y]}$$
Propagator term:
$$e^{\pm ik_zz}$$
}
